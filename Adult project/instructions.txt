# Calculate the percentage of people with advanced education (>50K)
advanced_education = ['Bachelors', 'Masters', 'Doctorate']
high_education = df[df.education.isin(advanced_education) & (df.salary == '>50K')]
percentage_high_education = (len(high_education) / len(df[df.education.isin(advanced_education)])) * 100

# Calculate the percentage of people without advanced education (>50K)
low_education = df[~df.education.isin(advanced_education) & (df.salary == '>50K')]
percentage_low_education = (len(low_education) / len(df[~df.education.isin(advanced_education)])) * 100

print("Percentage of people with advanced education making more than 50K:", percentage_high_education)
print("Percentage of people without advanced education making more than 50K:", percentage_low_education)
